### 2.2 动态SQL构建（数据抽取）

由于原始指标的数据源配置在 `ind_data_source` 表中，需要动态构建SQL：

```java
@Service
public class DataExtractService {

    public List<Map<String, Object>> extractRawData(IndDataSource source, LocalDate date, Long tenantId) {
        String sql = buildExtractSql(source, date, tenantId);
        // 使用动态数据源切换后执行
        return dynamicQuery(sql, source.getSourceDb());
    }

    private String buildExtractSql(IndDataSource source, LocalDate date, Long tenantId) {
        StringBuilder sql = new StringBuilder("SELECT ");
        // 日期字段
        sql.append(source.getDateField()).append(" as trade_date, ");
        // 时间点字段(如有)
        if (source.getTimeField() != null) {
            sql.append(source.getTimeField()).append(" as time_point, ");
        }
        // 值字段
        sql.append(source.getSourceField()).append(" as value, ");
        // 机组字段
        sql.append("unit_id ");
        sql.append("FROM ").append(source.getSourceTable());
        sql.append(" WHERE ").append(source.getDateField()).append(" = '").append(date).append("'");
        sql.append(" AND ").append(source.getTenantField()).append(" = ").append(tenantId);
        // 附加过滤条件
        if (source.getFilterCondition() != null) {
            JSONObject filter = JSON.parseObject(source.getFilterCondition());
            filter.forEach((k, v) -> sql.append(" AND ").append(k).append(" = '").append(v).append("'"));
        }
        return sql.toString();
    }
}
```

### 2.4 定时计算调度

```java
@Component
@Slf4j
public class IndicatorCalcJob {

    @Scheduled(cron = "0 0 2 * * ?")  // 每天凌晨2:00
    public void dailyCalc() {
        LocalDate calcDate = LocalDate.now().minusDays(1); // T+1
        Long tenantId = SecurityUtils.getTenantId(); // 多租户

        log.info("指标计算开始: date={}, tenant={}", calcDate, tenantId);

        // 1. 创建任务日志
        CalcTaskLog taskLog = createTaskLog(calcDate);

        try {
            // 2. 加载指标定义和公式
            List<IndicatorDef> indicators = defService.listAll(tenantId);
            List<CalcFormula> formulas = formulaService.listAll(tenantId);

            // 3. DAG排序
            DAGScheduler scheduler = new DAGScheduler(indicators, formulas);
            List<CalcLayer> layers = scheduler.resolveLayers();

            // 4. 逐层计算
            CalcExecutor executor = new CalcExecutor(
                this::loadRawData,     // 原始数据加载
                this::saveResult       // 结果持久化
            );
            CalcResult result = executor.execute(layers, calcDate, tenantId);

            // 5. 聚合(分时→日→月)
            aggregateService.aggregateDaily(calcDate, tenantId);
            aggregateService.aggregateMonthly(calcDate, tenantId);

            // 6. 更新Redis缓存
            refreshCache(calcDate, tenantId);

            // 7. 更新任务日志
            updateTaskLog(taskLog, result);

        } catch (Exception e) {
            log.error("指标计算失败", e);
            failTaskLog(taskLog, e.getMessage());
        }
    }
}
```

### 2.5 Redis缓存策略

```java
@Service
public class IndicatorCacheService {

    @Autowired
    private RedisTemplate<String, Object> redisTemplate;

    // 缓存Key设计
    // indicator:kpi:{tenantId}              → 首页5个KPI值
    // indicator:daily:{code}:{date}:{tenant} → 日维度数据
    // indicator:rank:{code}:{tenant}         → TOP10排名

    private static final Duration KPI_TTL = Duration.ofHours(1);
    private static final Duration DAILY_TTL = Duration.ofHours(6);
    private static final Duration RANK_TTL = Duration.ofHours(4);

    public void refreshKpiCache(Long tenantId) {
        // 查询5个核心指标的最新值
        List<String> kpiCodes = List.of(
            "MLT_CONTRACT_ENERGY", "ONLINE_ENERGY",
            "SPOT_FULL_FEE", "TOTAL_AVG_PRICE", "TOTAL_MARGIN"
        );
        Map<String, BigDecimal> kpiData = new HashMap<>();
        for (String code : kpiCodes) {
            DailyData latest = dataService.getLatest(code, tenantId);
            if (latest != null) kpiData.put(code, latest.getValue());
        }
        String key = "indicator:kpi:" + tenantId;
        redisTemplate.opsForHash().putAll(key, kpiData);
        redisTemplate.expire(key, KPI_TTL);
    }
}
```
